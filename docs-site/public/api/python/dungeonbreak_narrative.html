328def apply_one_event(state: dict[str, Any], event: Mapping[str, Any], snapshot: dict[str, Any] | None = None):
329    """
330    Apply one event in trait space.
331
332    Returns (new_state, log_row). Uses snapshot force data when available and
333    warns instead of inventing fallback force vectors.
334    """
335    snap = snapshot if snapshot is not None else load_narrative_snapshot()
336    traits = list(state.get("trait_names") or load_game_traits_manifest())
337
338    vector = np.array(state["vector"], float)
339    previous_vector = vector.copy()
340    total_xp = float(state.get("total_xp", 0.0))
341    effort_pool = float(state.get("effort_pool", 1.0))
342    cumulative_effort_spent = float(state.get("cumulative_effort_spent", 0.0))
343    warnings = list(state.get("warnings", []))
344
345    event_type = event.get("event_type", event.get("type", ""))
346    scene_id = str(event.get("scene_id", state.get("scene_id", SCENE_TOWN_SQUARE)))
347    chosen_dialog = event.get("chosen_dialog")
348
349    force_applied = False
350    dialog_can_apply_force = True
351    force_map, force_warning = _event_force_map(event, snap)
352
353    if event_type == EVENT_TRAINING:
354        total_xp += XP_GAIN_TRAINING
355    elif event_type == EVENT_REST:
356        effort_pool = min(1.0, effort_pool + EFFORT_RECOVERY_REST)
357    elif event_type == EVENT_DIALOG and chosen_dialog and effort_pool >= DEFAULT_EFFORT_COST_DIALOG:
358        effort_pool -= DEFAULT_EFFORT_COST_DIALOG
359        cumulative_effort_spent += DEFAULT_EFFORT_COST_DIALOG
360        total_xp += XP_GAIN_DIALOG
361    elif event_type == EVENT_DIALOG and chosen_dialog and effort_pool < DEFAULT_EFFORT_COST_DIALOG:
362        warnings.append("dialog_blocked_low_effort")
363        dialog_can_apply_force = False
364
365    # Apply snapshot-defined force only when available.
366    if force_map:
367        if event_type != EVENT_DIALOG or dialog_can_apply_force:
368            vector = vector + vector_from_trait_map(force_map, traits)
369            force_applied = True
370    elif force_warning:
371        warnings.append(force_warning)
372
373    effort_pool = min(1.0, max(0.0, effort_pool + EFFORT_RECOVERY_PER_EVENT))
374    level = 1 + int(total_xp // XP_PER_LEVEL)
375
376    quest_region = (snap.get("quests") or {}).get("main_ch1")
377    quest_ch1 = in_quest_region(vector, quest_region, traits) if quest_region else False
378    available = [label for label, _, _ in available_dialogs(vector, scene_id, snap, traits)]
379
380    new_state = {
381        "entity_name": state.get("entity_name", "Kaiza"),
382        "trait_names": traits,
383        "vector": vector.copy(),
384        "old_vector": previous_vector.copy(),
385        "total_xp": total_xp,
386        "effort_pool": effort_pool,
387        "cumulative_effort_spent": cumulative_effort_spent,
388        "scene_id": scene_id,
389        "data_available": bool(state.get("data_available", False) or force_applied),
390        "warnings": warnings,
391    }
392    log_row = {
393        "event_type": event_type,
394        "scene_id": scene_id,
395        "npc_id": SCENE_TO_NPC.get(scene_id),
396        "vector": vector.copy(),
397        "trait_map": trait_map_from_vector(vector, traits),
398        "level": level,
399        "total_xp": total_xp,
400        "effort_pool": effort_pool,
401        "cumulative_effort_spent": cumulative_effort_spent,
402        "quest_ch1_available": quest_ch1,
403        "available_options": available,
404        "chosen_dialog": chosen_dialog,
405        "force_applied": force_applied,
406        "warnings": list(warnings),
407    }
408    return new_state, log_row

199def available_dialogs(
200    entity_pos: np.ndarray,
201    scene_id: str | None = None,
202    snapshot: dict[str, Any] | None = None,
203    trait_names: list[str] | None = None,
204    threshold: float = DEFAULT_DIALOG_THRESHOLD,
205) -> list[tuple[str, float, dict[str, Any]]]:
206    """
207    Returns available dialogs as tuples of:
208    (label, distance, full_dialog_record).
209    """
210    traits = trait_names if trait_names is not None else load_game_traits_manifest()
211    snap = snapshot if snapshot is not None else load_narrative_snapshot()
212    dialog_vectors = _dialog_location_vectors(snap, traits)
213    in_range = options_above_threshold(np.array(entity_pos, float), dialog_vectors, threshold)
214
215    result: list[tuple[str, float, dict[str, Any]]] = []
216    for dialog, d in in_range:
217        allowed_scenes = dialog.get("scenes") or []
218        if scene_id and allowed_scenes and scene_id not in allowed_scenes:
219            continue
220        label = str(dialog.get("label", ""))
221        result.append((label, float(d), dialog))
222
223    result.sort(key=lambda row: row[1])
224    return result

288def in_quest_region(
289    vector: np.ndarray, region: dict[str, Any] | None, trait_names: list[str] | None = None
290) -> bool:
291    if not region:
292        return False
293    traits = trait_names if trait_names is not None else load_game_traits_manifest()
294    values = trait_map_from_vector(np.array(vector, float), traits)
295
296    for trait, min_value in (region.get("min") or {}).items():
297        if values.get(trait, 0.0) < float(min_value):

302    return True

 97def validate_game_alignment_warn_only(
 98    manifest_path: str | Path | None = None,
 99    snapshot_path: str | Path | None = None,
100    traits_dir: str | Path | None = None,
101) -> dict[str, Any]:
102    """
103    Compare folder trait assets vs manifest vs snapshot.
104
105    Warn-only behavior: returns warnings list but never raises.
106    """
107    folder_traits = _scan_folder_traits(Path(traits_dir) if traits_dir else None)
108    manifest_traits = load_game_traits_manifest(manifest_path)
109    snapshot = load_narrative_snapshot(snapshot_path)
110    snapshot_traits = [str(name) for name in snapshot.get("basis_vectors", [])]
111
112    warnings: list[str] = []
113
114    def _cmp(left: list[str], right: list[str], left_name: str, right_name: str) -> None:
115        left_only = sorted(set(left) - set(right))
116        right_only = sorted(set(right) - set(left))
117        if left_only:
118            warnings.append(f"{left_name} missing in {right_name}: {', '.join(left_only)}")
119        if right_only:
120            warnings.append(f"{right_name} extra vs {left_name}: {', '.join(right_only)}")
121        if left == right:
122            return
123        if not left_only and not right_only:
124            warnings.append(f"{left_name} and {right_name} contain same names but in different order")
125
126    _cmp(folder_traits, manifest_traits, "Folder traits", "Manifest traits")
127    _cmp(manifest_traits, snapshot_traits, "Manifest traits", "Snapshot basis_vectors")
128
129    return {
130        "ok": len(warnings) == 0,
131        "warnings": warnings,
132        "folder_traits": folder_traits,
133        "manifest_traits": manifest_traits,
134        "snapshot_traits": snapshot_traits,
135    }
